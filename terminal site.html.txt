        // Initialize application
        async function initializeApp() {
            // Check if user is already logged in
            const savedToken = localStorage.getItem('angelgotchi_token');
            const savedUser = localStorage.getItem('angelgotchi_user');

            if (savedToken && savedUser) {
                authToken = savedToken;
                currentUser = JSON.parse(savedUser);
                showProfileSection();
                await loadUserData();
                updateRecordsDisplay();
                updateTotalSteps();
                loadLeaderboard();
            } else {
                showAuthSection();
            }
        }

        // Authentication functions
        async function login() {
            const username = usernameInput.value.trim();
            const password = passwordInput.value.trim();

            if (!username || !password) {
                addToConsole('❌ Please enter both username and password');
                return;
            }

            try {
                addToConsole('🔄 Logging in...');

                const response = await fetch(`${API_BASE_URL}/login`, {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                    },
                    body: JSON.stringify({ username, password }),
                });

                const data = await response.json();

                if (response.ok) {
                    authToken = data.token;
                    currentUser = data.user;
                    localStorage.setItem('angelgotchi_token', authToken);
                    localStorage.setItem('angelgotchi_user', JSON.stringify(currentUser));

                    addToConsole('✅ Login successful! Welcome ' + username);
                    showProfileSection();
                    await loadUserData();
                    updateRecordsDisplay();
                    updateTotalSteps();
                    loadLeaderboard();
                } else {
                    addToConsole('❌ ' + (data.error || 'Login failed'));
                }
            } catch (error) {
                addToConsole('❌ Login failed: ' + error.message);
                // Fallback to local mode for development
                if (confirm('Server not available. Login locally for testing?')) {
                    currentUser = { id: 1, username: username };
                    userData = {
                        username: username,
                        records: JSON.parse(localStorage.getItem('angelgotchi_records_' + username) || '[]'),
                        totalSteps: 0
                    };
                    records = userData.records;
                    addToConsole('✅ Local login successful! Welcome ' + username);
                    showProfileSection();
                    updateRecordsDisplay();
                    updateTotalSteps();
                    loadLeaderboard();
                }
            }
        }

        async function register() {
            const username = usernameInput.value.trim();
            const password = passwordInput.value.trim();

                addToConsole('❌ Please enter both username and password');
                return;
            }

            if (username.length < 3) {
                addToConsole('❌ Username must be at least 3 characters');
                return;
            }

            if (password.length < 4) {
                addToConsole('❌ Password must be at least 4 characters');
                return;
            }

            try {
                addToConsole('🔄 Creating account...');

                const response = await fetch(`${API_BASE_URL}/register`, {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                    },
                    body: JSON.stringify({ username, password }),
                });

                const data = await response.json();

                if (response.ok) {
                    authToken = data.token;
                    currentUser = data.user;

        // Data management functions
        async function loadUserData() {
            if (!currentUser || !authToken) return;

            try {
                const response = await fetch(`${API_BASE_URL}/user`, {
                    headers: {
                        'Authorization': `Bearer ${authToken}`,
                    },
                });

                if (response.ok) {
                    userData = await response.json();
                    records = userData.records || [];
                } else {
                    throw new Error('Failed to load user data');
                }
            } catch (error) {
                addToConsole('📁 Loading data from local storage');
                // Fallback to local storage
                userData = {
                    username: currentUser.username,
                    records: JSON.parse(localStorage.getItem('angelgotchi_records_' + currentUser.username) || '[]'),
                    totalSteps: 0
                };
                records = userData.records;
            }
        }

        async function saveUserData() {
            if (!currentUser || !authToken) {
                // Save locally if no server connection
                localStorage.setItem('angelgotchi_records_' + currentUser.username, JSON.stringify(records));
                addToConsole('💾 Data saved locally');
                return;
            }

            // Try to save to server
            try {
                if (currentSteps > 0) {
                    const response = await fetch(`${API_BASE_URL}/steps`, {
                        method: 'POST',
                        headers: {
                            'Content-Type': 'application/json',
                            'Authorization': `Bearer ${authToken}`,
                        },
                        body: JSON.stringify({ steps: currentSteps }),
                    });

                    if (response.ok) {
                        addToConsole('✅ Steps saved to server');
                        // Reload user data to get updated totals
                        await loadUserData();
                    } else {
                        throw new Error('Failed to save to server');
                    }
                }
            } catch (error) {
                addToConsole('⚠️ Server save failed, saving locally');
                localStorage.setItem('angelgotchi_records_' + currentUser.username, JSON.stringify(records));
            }
        }

        async function loadLeaderboard() {
            try {
                const response = await fetch(`${API_BASE_URL}/leaderboard`);
                if (response.ok) {
                    const leaderboardData = await response.json();
                    displayLeaderboard(leaderboardData.users);
                } else {
                    throw new Error('Leaderboard not available');
                }
            } catch (error) {
                // Fallback: create leaderboard from local data
                addToConsole('📊 Generating leaderboard from local data');
                const localUsers = [];
                
                // Get all local user data
                for (let i = 0; i < localStorage.length; i++) {
                    const key = localStorage.key(i);
                    if (key.startsWith('angelgotchi_records_')) {
                        const username = key.replace('angelgotchi_records_', '');
                        const userRecords = JSON.parse(localStorage.getItem(key) || '[]');
                        const totalSteps = userRecords.reduce((total, record) => total + record.steps, 0);
                        localUsers.push({
                            username: username,
                            totalSteps: totalSteps
                        });
                    }
                }

                // Sort by total steps descending
                localUsers.sort((a, b) => b.totalSteps - a.totalSteps);
                displayLeaderboard(localUsers.slice(0, 10));
            }
        }

        }

        async function startReading() {
            if (!port) return;

            try {
                const decoder = new TextDecoder();
                reader = port.readable.getReader();

                while (true) {
                    const { value, done } = await reader.read();
                    if (done) break;

                    const data = decoder.decode(value);
                    processSerialData(data);
                }
            } catch (error) {
                addToConsole('❌ Reading error: ' + error.message);
            }
        }

        function processSerialData(data) {
            // Process incoming serial data from ESP32
            // This would parse step count data from your device
            const lines = data.split('\n');
            lines.forEach(line => {
                line = line.trim();
                if (line && !isNaN(line)) {
                    currentSteps = parseInt(line);
                    stepCount.textContent = currentSteps.toLocaleString();
                    addToConsole('📊 Current steps: ' + currentSteps);
                }
            });
        }

        async function readSteps() {
            if (!isConnected || !port) {
                addToConsole('❌ Device not connected');
                return;
            }

            try {
                addToConsole('📖 Reading steps from device...');
                // Send command to read steps from EEPROM
                const encoder = new TextEncoder();
                const writer = port.writable.getWriter();
                await writer.write(encoder.encode('READ_STEPS\n'));
                writer.releaseLock();

                // Wait a moment for response
                setTimeout(() => {
                    addToConsole('✅ Steps read from device');
                }, 1000);

            } catch (error) {
                addToConsole('❌ Read failed: ' + error.message);
            }
        }

            currentSteps = 0;
            stepCount.textContent = '0';
        }

        // Firmware flashing functions
        async function flashFirmware() {
            if (!flashFileInput.files.length) {
                addToConsole('❌ No firmware file selected');
                return;
            }

            const file = flashFileInput.files[0];
            const offset = parseInt(flashOffsetInput.value, 16);

            if (isNaN(offset)) {
                addToConsole('❌ Invalid flash offset');
                return;
            }

            flashBtn.disabled = true;
            eraseBtn.disabled = true;

            try {
                addToConsole('🔄 Starting firmware flash...');

                // Read file as array buffer
                const fileBuffer = await file.arrayBuffer();
                const fileData = new Uint8Array(fileBuffer);

                // Create flash options
                const flashOptions = {
                    fileArray: [{
                        data: String.fromCharCode(...fileData),
                        address: offset
                    }],
                    flashSize: "detect",
                    eraseAll: false,
                    compress: true,
                    reportProgress: (fileIndex, written, total) => {
                        const progress = ((written / total) * 100).toFixed(1);
                        addToConsole(`📊 Flashing progress: ${progress}% (${written}/${total} bytes)`);
                    }
                };

                await esploader.writeFlash(flashOptions);
                addToConsole('✅ Firmware flashed successfully!');

            } catch (error) {
                addToConsole('❌ Flash failed: ' + error.message);
            } finally {
                flashBtn.disabled = false;
                eraseBtn.disabled = false;
            }
        }

                        port = availablePorts[0];

                        addToConsole('✅ Device connected successfully');
                        addToConsole('🔄 Starting data reader...');

                        // Initialize esptool-js
                        const { ESPLoader, Transport } = window;
                        transport = new Transport(port, true);
                        esploader = new ESPLoader({
                            transport: transport,
                            baudrate: 115200,
                            terminal: {
                                clean: () => {},
                                writeLine: (data) => addToConsole('ESPTOOL: ' + data),

                            }
                        });

                        // Hide troubleshooting on successful connection
                        hideTroubleshooting();

                        // Start reading data
                        startReading();
                        return;

                    } catch (existingPortError) {
                        addToConsole('⚠️ Previously used device not available: ' + existingPortError.message);
                        addToConsole('🔄 Requesting new device selection...');
                    }
                }

                updateStatus('connecting', 'Connecting...', 'Please select your AngelGotchi device from the list...');

                // Request a new port
                port = await navigator.serial.requestPort();

                // Open the port
                await port.open({
                    baudRate: 115200,
                    dataBits: 8,
                    stopBits: 1,
                    parity: 'none'
                });

                isConnected = true;
                updateStatus('connected', 'Device Connected', 'Successfully connected to AngelGotchi device');

                // Enable buttons
                connectBtn.disabled = true;
                disconnectBtn.disabled = false;
                readBtn.disabled = false;
                recordBtn.disabled = false;
                flashBtn.disabled = false;
                eraseBtn.disabled = false;

                addToConsole('✅ Device connected successfully');
                addToConsole('🔄 Starting data reader...');

                // Initialize esptool-js
                const { ESPLoader, Transport } = window;
                transport = new Transport(port, true);
                esploader = new ESPLoader({
                    transport: transport,
                    baudrate: 115200,
                    terminal: {
                        clean: () => {},
                        writeLine: (data) => addToConsole('ESPTOOL: ' + data),
                        write: (data) => {}
                    }
                });

                addToConsole('🔧 ESPTool initialized');

                // Hide troubleshooting on successful connection
                hideTroubleshooting();

                // Start reading data
                startReading();

            } catch (error) {
                console.error('Connection error:', error);

                if (error.name === 'NotAllowedError') {
                    updateStatus('disconnected', 'Permission Denied', 'Please allow access to serial devices and try again');
                    addToConsole('❌ Permission denied - Please click "Allow" when prompted for device access');
                    addToConsole('💡 Make sure no other programs are using the serial port');

                addToConsole('   2. Unplug and replug the USB-C cable');

        async function startReading() {
            try {
                while (port && port.readable) {
                    reader = port.readable.getReader();
                    const decoder = new TextDecoder();

                    try {
                        while (true) {
                            const { value, done } = await reader.read();
                            if (done) break;

                            const chunk = decoder.decode(value);
                            processSerialData(chunk);
                        }
                    } catch (error) {
                        addToConsole('⚠️ Reader error: ' + error.message);
                    } finally {
                        reader.releaseLock();
                    }
                }
            } catch (error) {
                addToConsole('❌ Reading error: ' + error.message);
            }
        }

        function processSerialData(data) {
            // Add raw data to console
            const lines = data.split('\n');
            lines.forEach(line => {
                if (line.trim()) {
                    addToConsole('📥 ' + line.trim());
                }
            });

            // Try to extract step count from the data
            const stepMatch = data.match(/Steps?:\s*(\d+)/i);
            if (stepMatch) {
                const steps = parseInt(stepMatch[1]);
                if (!isNaN(steps)) {
                    currentSteps = steps;
                    stepCount.textContent = steps.toLocaleString();
                    addToConsole('📊 Step count updated: ' + steps);
                }
            }

            // Look for other useful data patterns
            if (data.includes('EEPROM') || data.includes('steps')) {
                addToConsole('💾 EEPROM/Step data detected');
            }
        }
